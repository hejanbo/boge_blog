
-- 创建数据库脚本
```sql

-- 服务器登陆
sudo -u postgres psql
-- 创建用户boge, 密码为123456
CREATE USER boge WITH PASSWORD '123456';
-- 创建数据库
CREATE DATABASE boge_blog;
-- 将数据库boge_blog权限全部给用户boge
GRANT ALL PRIVILEGES ON DATABASE boge_blog TO boge;
-- 创建public权限
\c boge_blog
GRANT ALL ON SCHEMA public TO boge;
-- 退出pg库
\q

-- 测试是否能能正常连接
psql -U boge -d boge_blog -h 127.0.0.1
```

删除数据库
```sql
sudo -u postgres psql

DROP DATABASE IF EXISTS boge_blog;
REVOKE ALL ON SCHEMA public FROM boge;
DROP USER IF EXISTS boge;
```
